#include <bits/stdc++.h>
using namespace std;
int main() {
    int n,value;
    cin >> n;
    int arr[n];
    for (int i=0;i<n;i++){
        cin >> arr[i];
    }
    cin >> value;
    for (int i=0;i<n;i++){
        if (arr[i]==value){
            cout << i << endl;
            break;
        }
        else if (i==(n-1)){
            cout << "-1" << endl;
        }
    }
    return 0;
}
