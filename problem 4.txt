#include <bits/stdc++.h>
using namespace std;

int main() {
    int r,c,count=0;
    cin >> r >> c;
    int justice_league[r][c], villains[r][c];
    for (int i = 0; i < r; i++) {
        for (int j = 0; j < c; j++){
            cin >> justice_league[i][j];
    }}
    for (int i = 0; i < r; i++) {
        for (int j = 0; j < c; j++){
            cin >> villains[i][j];
    }}
    for (int i = 0; i < r; i++) {
        for (int j = 0; j < c; j++){
            if (justice_league[i][j] > villains[i][j]){
               count++; 
            }
            else if (justice_league[i][j] < villains[i][j]){
                count--;
            }
        }   
    }
    if (count>0){
        cout << "Justice League";
    }
    else if (count<0){
        cout << "Villains";
    }
    else {
        cout << "Tie";
    }
    return 0;
}
